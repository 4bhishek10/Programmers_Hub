Bresenham Circle Class

#include<iostream>

#include<graphics.h>

#include<math.h>

using namespace std;

class a

{

public:

int xc,yc,r;

void input()

{

cout<<"Enter radius";

cin>>r;

cout<<"Enter xc";

cin>>xc;

cout<<"Enter yc";

cin>>yc;

}

};

class b:public a

{

public:

void circle(int xc,int yc,int r)

{

int x,y;

x=0;

y=r;

int p0,pk;

p0=3-(2*r);

pk=p0;

while(x<=y)

{

if(pk<0)

{

x=x+1;

pk=pk+4*x+6;

}

else

{

x=x+1;

y=y-1;

pk=pk+4*(x-y)+10;

}

putpixel(xc+x,yc+y,RED);

putpixel(xc+y,yc+x,RED);

putpixel(xc+-x,yc+-y,RED);

putpixel(xc+-y,yc+-x,RED);

putpixel(xc+-x,yc+y,RED);

putpixel(xc+-y,yc+x,RED);

putpixel(xc+x,yc+-y,RED);

putpixel(xc+y,yc+-x,RED);

}

}

};

int main()

{

b o;

o.input();

int gd=DETECT,gm;

initgraph(&gd,&gm,NULL);

o.circle(o.xc,o.yc,o.r);

delay(5000);

closegraph;

}
