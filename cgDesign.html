#include<iostream>

#include<graphics.h>

#include<math.h>

using namespace std;

void DDA(float ,float,float,float);

int main()

{

int gd=DETECT,gm;

float x1,y1,x2,y2,xm,ym,xm1,ym1,xp,yp,xq,yq,xr,yr,xs,ys;

cout<<"Enter co-ordinates x1 and y1";

cin>>x1>>y1;

cout<<"Enter co-ordinates x2 and y2";

cin>>x2>>y2;

xm=(x1+x2)/2;

ym=(y1+y2)/2;

xm1=(x1+x2)/2;

ym1=(y1+y2)/2;

xp=(x1+xm)/2;

yp=(y1+ym1)/2;

xq=(x1+xm1)/2;

yq=(ym1+y2)/2;

xr=(xm1+x2)/2;

yr=(y2+ym)/2;

xs=(x2+xm)/2;

ys=(ym+y1)/2;

initgraph(&gd,&gm,NULL);

DDA(x1,x2,y1,y1);

DDA(x1,x2,y2,y2);

DDA(x2,x2,y2,y1);

DDA(x1,x1,y1,y2);

DDA(xm,x1,y1,ym1);

DDA(x1,xm1,ym1,y2);

DDA(xm1,x2,y2,ym);

DDA(x2,xm,ym,y1);

DDA(xp,xq,yp,yq);

DDA(xq,xr,yq,yr);

DDA(xr,xs,yr,ys);

DDA(xp,xs,yp,ys);

closegraph;

return 0;

}

void DDA(float x1,float x2,float y1,float y2)

{

float dx,dy,x,y,xinc,yinc,length;

dx=x2-x1;

dy=y2-y1;

if(abs(dx)>abs(dy))

length=abs(dx);

else

length=abs(dy);

xinc=dx/length;

yinc=dy/length;

x=x1;

y=y1;

for(int v=0;v<=length;v++)

{

x=x+xinc;

y=y+yinc;

putpixel(round(x),round(y),GREEN);

//DDA(x1,y1,x2,y2);

delay(10);

}

}
