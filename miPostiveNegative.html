section .data

space db " "

spacelength equ $-space

pcnt db 0

ncnt db 0

section .bss

disp_buff resb 2

%macro write 2

mov rax,1

mov rdi,1

mov rsi,%1

mov rdx,%2

syscall

%endmacro

section .text

global _start

_start:

mov rsi,array

mov rcx,02

l3:

mov bl,[rsi]

call disp_proc

rol bl,01

jc l5

inc byte[pcnt]

jmp l6

l5:

inc byte[ncnt]

l6:

inc rsi

dec rcx

jnz l3

mov bl,[pcnt]

call disp_proc

mov bl,[ncnt]

call disp_proc

mov rax,60

mov rdi,0

syscall

disp_proc:

push rcx

push rdi

push rsi

mov rdi,disp_buff

mov rcx,02

l1: rol bl,04

mov dl,bl

and dl,0fh

cmp dl,09h

jbe l2

add dl,07h

l2: add dl,30h

mov [rdi],dl

inc rdi

dec rcx

jnz l1

write disp_buff,2

write space,spacelength

pop rsi

pop rdi

pop rcx

ret
