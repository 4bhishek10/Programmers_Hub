Hex To Bcd

section .data

num db 02Ch

cnt db 00

msg db 'HEX no. is',10

len equ $-msg

msg2 db 'No. in BCD'

len2 equ $-msg2

section .bss

disbuff resb 02

result resb 02

%macro disp 2

mov rax,1

mov rdi,1

mov rsi,%1

mov rdx,%2

syscall

%endmacro

section .text

global _start

_start:

disp msg,len

mov bl,byte[num]

call ascii

disp msg2,len2

call proc1

mov rax,60

mov rdi,0

syscall

ascii:

push rcx

push rdi

push rsi

mov rcx,02

mov rdi,disbuff

l2:rol bl,04

mov dl,bl

and dl,0fh

cmp dl,09h

jbe l1

add dl,07h

l1:add dl,30h

mov [rdi],dl

inc rdi

dec rcx

jnz l2

disp disbuff,02

pop rsi

pop rdi

pop rcx

ret

proc1:

mov bl,byte[num]

call ascii

mov rcx,0

mov al,bl

mov bx,10

l3:mov dx,00

div bx

push rdx

inc rcx

inc byte[cnt]

cmp ax,00

jne l3

disp msg2,len2

l4:pop rdx

add dl,30h

mov [result],dl

disp result,2

dec byte[cnt]

jnz l4

ret
