XYShear And rotation

#include<iostream>

#include<graphics.h>

#include<math.h>

using namespace std;

int main()

{

int gd=DETECT;

int gm,x1,y1,x2,y2,x3,y3,xsh,ysh,xsh1,ysh1,xsh2,ysh2,xsh3,ysh3;

x1 = 100;

y1 = 150;

x2 =120;

y2 = 200;

x3 = 50;

y3 = 200;

int ch;

cout<<"Operations to be performed \n 1.x shear \n 2.y shear";

cout<<"\n 3.Reflection about x-axis \n 4.Reflection about y-axis ";

cin>>ch;

switch(ch)

{

case 1 :

cout<<" \n Enter x shear value";

cin>>xsh;

xsh1 = x1+xsh*y1;

xsh2 = x2+xsh*y2;

xsh3 = x3+xsh*y3;

initgraph(&gd,&gm,NULL);

delay(20);

setcolor(WHITE);

line(x1,y1,x2,y2);

line(x2,y2,x3,y3);

delay(20);

line(x3,y3,x1,y1);

setcolor(4);

line(xsh1,y1,xsh2,y2);

line(xsh2,y2,xsh3,y3);

delay(20);

line(xsh3,y3,xsh1,y1);

delay(1000);

closegraph();

break;

case 2:

cout<<" \n Enter y shear value";

cin>>ysh;

ysh1 = y1+ysh*x1;

ysh2 = y2+ysh*x2;

ysh3 = y3+ysh*x3;

initgraph(&gd,&gm,NULL);

setcolor(WHITE);

line(x1,y1,x2,y2);

delay(20);

line(x2,y2,x3,y3);

line(x3,y3,x1,y1);

setcolor(4);

line(x1,ysh1,x2,ysh2);

delay(20);

line(x2,ysh2,x3,ysh3);

line(x3,ysh3,x1,ysh1);

delay(1500);

closegraph();

break;

case 3:

cout<<"Relfection about x-axis";

initgraph(&gd,&gm,NULL);

int maxx,maxy;

maxx=getmaxx();

maxy=getmaxy();

float midx,midy;

midx=maxx/2;

midy=maxy/2;

line(0,midy,maxx,midy);

outtextxy(0,midy,"X-axis");

delay(500);

line(midx,0,midx,maxy);

outtextxy(midx,0,"Y-axis");

delay(1000);

line(midx+x1,y1,midx+x2,y2);

line(midx+x2,y2,midx+x3,y3);

line(midx+x3,y3,midx+x1,y1);

line(midx+x1,maxy-y1,midx+x2,maxy-y2);

line(midx+x2,maxy-y2,midx+x3,maxy-y3);

line(midx+x3,maxy-y3,midx+x1,maxy-y1);

delay(5000);

closegraph();

break;

case 4:

cout<<"Relfection about y-axis";

initgraph(&gd,&gm,NULL);

int maxx1,maxy1;

maxx1=getmaxx();

float midx1,midy1;

maxy1=getmaxy();

midx1=maxx1/2;

midy1=maxy1/2;

line(0,midy1,maxx1,midy1);

outtextxy(0,midy1,"X-axis");

delay(500);

line(midx1,0,midx1,maxy1);

outtextxy(midx1,0,"Y-axis");

delay(1000);

line(midx1+x1,y1,midx1+x2,y2);

line(midx1+x2,y2,midx1+x3,y3);

line(midx1+x3,y3,midx1+x1,y1);

line(midx1-x1,y1,midx1-x2,y2);

line(midx1-x2,y2,midx1-x3,y3);

line(midx1-x3,y3,midx1-x1,y1);

delay(5000);

closegraph();

break;

}

return(0);

}
