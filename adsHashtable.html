#include<string.h>

#define h(x) x%10

using namespace std;

struct linear

{

string name;

}l[10];

class lin_probing

{

int loc;

lin_probing()

{

for(int i=0;i<10;i++)

{

l[i].mob_no=-1;

}

}

void insert(string,long int,int);

void display();

};

void lin_probing::insert(string nam,long int no,int i)

{

loc=h(no);

if(l[loc].mob_no==-1)

{

l[loc].name=nam;

l[loc].mob_no=no;

}

else

{

int i=loc+1;

i=i%10;

while(i!=loc)

{

if(l[i].mob_no==-1)

{

l[i].name=nam;

l[i].mob_no=no;

break;

}

i++;

}

}

}

void lin_probing::display()

{

cout<<"\nName:\tNumber:\n";

for(int i=0;i<10;i++)

{

cout<<l[i].name<<"\t"<<l[i].mob_no<<"\n";

}

}

void lin_probing::search(long int number)

{

loc=h(number);

if(l[loc].mob_no==number)

{

cout<<"\nData found\t";

}

else

{

int i=loc+1;

i=(loc+1)%10;

{

if(l[i].mob_no==number)

{

}
