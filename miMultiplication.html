Multiplication

section .data

mul1 db 10

mul2 db 01

msg1 db 'multiplicand is ' ,10

len1 equ $-msg1

msg2 db 'multiplier is ',10

len2 equ $-msg2

msg3 db 'multiplication by successive addition is',10

len3 equ $-msg3

msg4 db 'multiplication by shift and add is',10

len4 equ $-msg4

section .bss

disbuff resb 2

%macro disp 2

mov rax,1

mov rdi,1

mov rsi,%1

mov rdx,%2

syscall

%endmacro

section .text

global _start

_start:

disp msg1,len1

mov bl,byte[mul1]

call ascii

disp msg2,len2

mov bl,byte[mul2]

call ascii

call proc1

disp msg3,len3

call proc2

mov rax,60

mov rdi,0

syscall

ascii:

push rcx

push rdi

push rsi

mov rdi,disbuff

mov rcx,02

l2:rol bl,04

mov dl,bl

and dl,0fh

cmp dl,09h

jbe l1

add dl,07h

l1:add dl,30h

mov [rdi],dl

inc rdi

dec rcx

jnz l2

disp disbuff,02

pop rsi

pop rdi

pop rcx

ret

proc1:

push cx

mov ax,0000

mov bl,byte[mul1]

mov cl,byte[mul2]

l4:add al,bl

jnc l3

inc ah

l3:dec cl

jnz l4

mov bx,ax

disp msg3,len3

call ascii

pop cx

ret

proc2:

mov bx,0000

mov cl,08

mov ah,0

mov bl,byte[mul1]

mov al,byte[mul2]

l6:shr al,01

jnc l5

add bx,dx

l5:shl dx,01

dec cl

jnz l6

disp msg4,len4

call ascii

ret

proc2:

push rcx

mov ax,0000

mov cl,08

mov bl,byte[mul1]

mov dl,byte[mul2]

l7:shr dl,01

jnc l6

add ax,bx

l6:shl bl,01

dec cl

jnz l7

mov bx,ax

call ascii

pop rcx

ret
